void liberar_historial(historial_t *registro) {
    nodo_historial_t *actual = registro->cabeza;
    while (actual != NULL) {
        nodo_historial_t *temp = actual;
        actual = actual->next;
        free(temp->paciente);  // ✅ Liberar el paciente
        free(temp);            // ✅ Liberar el nodo
    }
    free(registro);
}

void liberar_pila(pila_t* pila) {

7 Iteración insegura durante el triage dinámico ▼
programa.c Líneas 129–158

El código avanza el iterador actual = actual->next antes de extraer, lo cual evita el caso más obvio. Sin embargo, si otro paciente posterior en la misma cola fue extraído en una iteración anterior (dentro del mismo bucle), el puntero actual->next puede apuntar a memoria liberada.

✅ Solución — Usar lista de cambios pendientes
// En vez de modificar la cola DURANTE la iteración,
// recolectar los cambios y aplicarlos DESPUÉS:

typedef struct cambio_pendiente {
    int id_paciente;
    int cola_origen;   // índice 0-3
    int nueva_prio;
} cambio_pendiente_t;

// 1. Iterar y recolectar cambios (sin modificar colas)
// 2. Aplicar todos los cambios después del bucle
for (int c = 0; c < total_cambios; c++) {
    patient_t *extraido = extraer_paciente(
        colas[cambios[c].cola_origen],
        cambios[c].id_paciente
    );
    if (extraido) {
        push(pila_rollback, extraido->id, extraido->prioridad_actual);
        extraido->prioridad_actual = cambios[c].nueva_prio;
        encolar(colas[cambios[c].nueva_prio - 1], extraido);
    }
}
Nota: Puedes usar una cola o lista enlazada para los cambios pendientes (no un arreglo) y así cumplir con la restricción de no usar arreglos.

void imprimir_snapshot(int tick, espera_t** colas,
